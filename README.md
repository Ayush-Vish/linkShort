</ol>

</ol>
<pre><code>cd url-shortener</code></pre>

</ol>
<pre><code>npm install
# or

<ol start="4">

<pre><code>DATABASE_URL: The URL for your database, e.g., a MongoDB connection string.

<ol start="5">

</ol>
<pre><code>npm start
# or
yarn start</code></pre>
